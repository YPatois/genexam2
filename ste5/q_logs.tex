\element{logs}{
    \begin{question}{logs1}
        Le logarithme décimal de $10^{-3}$ est :
        \begin{reponses}
            \mauvaise{$3$}
            \mauvaise{$0$}
            \mauvaise{$-1$}
            \bonne{$-3$}
        \end{reponses}
    \end{question}
}

\element{logs}{
    \begin{question}{logs2}
        Si $\log x = 2$, alors $x$ est :
        \begin{reponses}
            \mauvaise{$0{,}01$}
            \mauvaise{$0{,}1$}
            \mauvaise{$1$}
            \bonne{$100$}
        \end{reponses}
    \end{question}
}

\element{logs}{
    \begin{question}{logs3}
        $\log(ab)$ est égal à :
        \begin{reponses}
            \mauvaise{$\log a - \log b$}
            \mauvaise{$\log a \times \log b$}
            \mauvaise{$\frac{\log a}{\log b}$}
            \bonne{$\log a + \log b$}
        \end{reponses}
    \end{question}
}

\element{logs}{
