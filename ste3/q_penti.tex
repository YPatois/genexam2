
\element{penti}{
  \begin{question}{penti1}
    Quelle est la valeur de \(2^3\)~?
    \begin{reponses}
      \mauvaise{$6$}
      \mauvaise{$4$}
      \mauvaise{$10$}
      \bonne{$8$}
    \end{reponses}
    \explain{\(2^3 = 2 \times 2 \times 2 = 8\)}
  \end{question}
}
